The expected directory path should be ~/yahboomcar_ws/src/simple_maze_bot, inside the directory, you can see files below

#Let's start!
cd ~/yahboomcar_ws

# assembly your system
colcon build

# if you modify any files in this dir, please assembly again
colcon build --packages-select simple_maze_bot

# delete build
rm -rf build/ install/

# execute every time you open a terminal
source install/setup.bash

# launch phase 1 test
ros2 launch simple_maze_bot sim_phase1_test.launch.py
# launch phase 2 test
ros2 launch simple_maze_bot sim_phase2_test.launch.py


#open a new terminal, start navigation
ros2 run simple_maze_bot go_to_start

ros2 run simple_maze_bot start_maze_task

# launch RViz
rviz2 -d $(ros2 pkg prefix nav2_bringup)/share/nav2_bringup/rviz/nav2_default_view.rviz



# In case you cannot find executables, try to debug with this
ros2 pkg executables simple_maze_bot
